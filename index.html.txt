        function showTab(id) {
            document.querySelectorAll('.content-section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(id).classList.add('active');

        }

        function init() {
            days.forEach(d => {
                const h = document.createElement('div');
                h.className = 'grid-header'; h.innerText = d;
                grid.appendChild(h);
            });

            for (let hour = 6; hour <= 24; hour++) {
                const label = document.createElement('div');
                label.className = 'time-cell'; label.innerText = `${hour.toString().padStart(2, '0')}:00`;
                grid.appendChild(label);
                for (let d = 1; d <= 7; d++) {
                    const slot = document.createElement('div');
                    slot.className = 'slot'; slot.id = `slot-${d}-${hour}`;
                    slot.onclick = () => cycleSlot(slot);
                    grid.appendChild(slot);
                }
            }
            loadData();
            calcTotal();
        }

        function cycleSlot(el) {
            const types = ['', 'slot-strategic', 'slot-working', 'slot-buffer', 'slot-breakout'];
            const names = ['', '策略', '工作', '緩衝', '抽離'];
            let idx = 0;
            types.forEach((t, i) => { if(t && el.classList.contains(t)) idx = i; });
            let next = (idx + 1) % types.length;
            el.className = 'slot ' + types[next];
            el.innerText = names[next];
            localStorage.setItem('slot-'+el.id, JSON.stringify({txt: el.innerText, cls: el.className}));
        }

        function calcTotal() {
            let total = 0;
            let count = 0;
            for(let i=1; i<=5; i++) {
                let val = document.getElementById(`g1-c${i}`).value;
                let num = parseFloat(val) || 0;
                total += num;
                count++;
            }
            document.getElementById('g1-total').innerText = Math.round(total / count) + "%";
        }

        function applyQuickInput() {
            const val = document.getElementById('quickInput').value.trim();
            const regex = /([一二三四五六日])\s*(\d{1,2}):\d{2}-(\d{1,2}):\d{2}\s*(.*)/;
            const match = val.match(regex);
            if (match) {
                const dMap = {'一':1,'二':2,'三':3,'四':4,'五':5,'六':6,'日':7};
                const d = dMap[match[1]], sH = parseInt(match[2]), eH = parseInt(match[3]), txt = match[4];
                let cls = 'slot-working';
                if(txt.includes('策略')) cls = 'slot-strategic';
                else if(txt.includes('休息') || txt.includes('抽離')) cls = 'slot-breakout';
                else if(txt.includes('緩衝')) cls = 'slot-buffer';
                for (let h = sH; h < eH; h++) {
                    const el = document.getElementById(`slot-${d}-${h}`);
                    if (el) { el.innerText = txt; el.className = 'slot ' + cls; localStorage.setItem('slot-'+el.id, JSON.stringify({txt:txt, cls:el.className})); }
                }
                document.getElementById('quickInput').value = '';
            } else { alert("格式範例：一 09:00-11:00 策略塊"); }
        }

        function saveTextData() {
            document.querySelectorAll('textarea, input[type="text"]').forEach(el => {
                localStorage.setItem('text-' + el.id, el.value);
            });
        }

        function loadData() {
            document.querySelectorAll('textarea, input[type="text"]').forEach(el => {
                const saved = localStorage.getItem('text-' + el.id);
                if (saved) el.value = saved;
                el.oninput = () => { saveTextData(); if(el.id.includes('-c')) calcTotal(); };
            });
            for (let h = 6; h <= 24; h++) {
                for (let d = 1; d <= 7; d++) {
                    const saved = localStorage.getItem(`slot-slot-${d}-${h}`);
                    if (saved) {
                        const obj = JSON.parse(saved);
                        const el = document.getElementById(`slot-${d}-${h}`);
                        el.innerText = obj.txt; el.className = obj.cls;
                    }
                }
            }
        }
